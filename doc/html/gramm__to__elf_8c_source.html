157  int cmpt_octet = 0;
158  int i = 0; int n, cmpt_inter,j ;
159  int taille_data_prog = (int)((*p_taille_data)/4);
160  int data_prog[taille_data_prog]; /*EN OCTETS*/
161  memset(data_prog, 0x0, *p_taille_data);
162  DATA data_courant = NULL;
163 
164  /*On raisonne sur les octets*/
165  if((*p_taille_data)>0){
166 
167  while((*p_file_Data) != NULL){
168  data_courant = (DATA)((*p_file_Data)->suiv->val);
169  switch(data_courant->type){
170  case DATA_BYTE :
171  /*On trouve .byte n => Il faut faire 1 OCTETS de valeur "n" dans data_prog */
172  while(data_courant->l_operande != NULL){
173  n = ((OPERANDE)((data_courant->l_operande)->val))->bin;
174 
175  data_prog[i] = data_prog[i] | deplace_octet(n, 3-cmpt_octet%4);
176  cmpt_octet++; i = (int)(cmpt_octet/4);
177 
178  (data_courant->l_operande) = (data_courant->l_operande)->suiv;
179  }
180  break;
181  break;
182 
183  case DATA_SPACE :
184  /*On trouve .space n => Il faut faire n OCTETS nuls dans data_prog */
185  while(data_courant->l_operande != NULL){
186  n = ((OPERANDE)((data_courant->l_operande)->val))->bin;
187  cmpt_inter = cmpt_octet + n;
188  for(;cmpt_octet<cmpt_inter; cmpt_octet++){
189  data_prog[i] = data_prog[i] | deplace_octet(0x0, 3-cmpt_octet%4);
190  cmpt_octet++; i = (int)(cmpt_octet/4);
191  }
192  (data_courant->l_operande) = (data_courant->l_operande)->suiv;
193  }
194  break;
195 
196  case DATA_ASCIIZ :
197  /*On trouve .ascizz "chaine" => il faut remplir un octet par caractere de la chaine*/
198  while(data_courant->l_operande != NULL){
199  n = strlen( ((OPERANDE)((data_courant->l_operande)->val))->chain ) - 1;
200  char chaine[n];
201  strcpy(chaine, ((OPERANDE)((data_courant->l_operande)->val))->chain + 1); /*On enleve la premiere quote*/
202  chaine[n] = '\0';
203  cmpt_inter = n + i;
204  for(j =0;j<=n; j++){
205  data_prog[i] = data_prog[i] | deplace_octet(chaine[j], 3-cmpt_octet%4);
206  cmpt_octet++; i = (int)(cmpt_octet/4);
207  }
208

210  }

212

217 
218  if(cmpt_octet%4!=0 && cmpt_octet != 0){
219  i++; cmpt_octet += 4 - i%4;
220  }
221  data_prog[i] = n;
222  data_prog[i] = ((data_prog[i]>>24) & 0x000000ff) | // move byte 3 to byte 0
223  ((data_prog[i]<<8) & 0x00ff0000) | // move byte 1 to byte 2
224  ((data_prog[i]>>8) & 0x0000ff00) | // move byte 2 to byte 1
225  ((data_prog[i]<<24) & 0xff000000); // byte 0 to byte 3
226 
227  i++; cmpt_octet+=4;
228 
229  (data_courant->l_operande) = (data_courant->l_operande)->suiv;
230  }
231  break;

235  break;
236  }

239  }
